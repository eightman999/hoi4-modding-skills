# Toggle Window Pattern Template
# Creates a window that can be opened/closed with a button
# Placeholders:
# - SYSTEM_NAME: Your system/feature name
# - WINDOW_TITLE: Window title localisation key

### SCRIPTED GUI FILE (common/scripted_guis/bsm_SYSTEM_NAME_sgui.txt) ###

scripted_gui = {
	bsm_SYSTEM_NAME_toggle = {
		context_type = player_context
		window_name = "bsm_SYSTEM_NAME_toggle_container"

		visible = {
			has_country_flag = SYSTEM_NAME_enabled
		}

		effects = {
			toggle_window_button_click = {
				if = {
					limit = { has_variable = SYSTEM_NAME_window_open }
					clear_variable = SYSTEM_NAME_window_open
				}
				else = {
					set_variable = { SYSTEM_NAME_window_open = 1 }
				}
			}

			close_button_click = {
				clear_variable = SYSTEM_NAME_window_open
			}
		}

		triggers = {
			window_content_visible = {
				has_variable = SYSTEM_NAME_window_open
			}
		}

		ai_enabled = {
			always = no
		}
	}
}

### GUI FILE (interface/bsm_SYSTEM_NAME.gui) ###

guiTypes = {

		size = { width = 100%% height = 100%% }

			name = "toggle_window_button"
