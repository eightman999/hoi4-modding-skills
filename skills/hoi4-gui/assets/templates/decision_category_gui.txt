# Decision Category GUI Pattern Template
# Creates a GUI integrated with a decision category
# Placeholders:
# - SYSTEM_NAME: Your system name
# - CATEGORY_NAME: Decision category name

### DECISION CATEGORY FILE (common/decisions/categories/bsm_SYSTEM_NAME_categories.txt) ###

bsm_SYSTEM_NAME_category = {
	icon = GFX_decision_category_SYSTEM_NAME

	visible = {
		has_country_flag = SYSTEM_NAME_enabled
	}

	allowed = {
		always = yes
	}

	visible_when_empty = yes
	priority = 50

	# Link to scripted GUI
	scripted_gui = bsm_SYSTEM_NAME_gui
}

### SCRIPTED GUI FILE (common/scripted_guis/bsm_SYSTEM_NAME_sgui.txt) ###

scripted_gui = {
	bsm_SYSTEM_NAME_gui = {
		context_type = decision_category
		window_name = "bsm_SYSTEM_NAME_gui"

		visible = {
			has_country_flag = SYSTEM_NAME_enabled
		}

		effects = {
			main_action_button_click = {
				# Your main action
				add_political_power = -25
				custom_effect = yes
			}

			option_1_button_click = {
				# Option 1 effect
			}

			option_2_button_click = {
				# Option 2 effect
			}
		}

		triggers = {
			main_action_button_click_enabled = {
				has_political_power > 24
			}

			option_1_button_visible = {
				has_country_flag = option_1_available
			}

			option_2_button_visible = {
				has_country_flag = option_2_available
			}
		}

		properties = {
			status_icon = {
				frame = status_value  # Variable controlling icon frame
			}
		}

		ai_enabled = {
			always = no
		}
	}
}
