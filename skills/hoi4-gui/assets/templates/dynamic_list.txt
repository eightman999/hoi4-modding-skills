# Dynamic List Pattern Template
# Creates a scrollable list populated from an array
# Placeholders:
# - SYSTEM_NAME: Your system name
# - LIST_NAME: Name of the list (e.g., members, countries, items)
# - ARRAY_NAME: Variable array name

### SCRIPTED GUI FILE (common/scripted_guis/bsm_SYSTEM_NAME_sgui.txt) ###

scripted_gui = {
	bsm_SYSTEM_NAME_list = {
		context_type = player_context  # Or decision_category
		window_name = "bsm_SYSTEM_NAME_list"

		visible = {
			has_country_flag = SYSTEM_NAME_enabled
		}

		effects = {
			# Button in each list entry
			item_button_click = {
				# Access item via 'v' (value) and 'i' (index) variables
				set_variable = { selected_index = i }
				set_temp_variable = { selected_value = v }

				# Your action here
				custom_effect = yes
			}
		}

		triggers = {
			item_button_click_enabled = {
				# Condition for item to be clickable
				always = yes
			}

			# Highlight selected item
			item_highlight_visible = {
				check_variable = { selected_index = i }
			}
		}

		properties = {
			# Dynamic icon based on array value
			item_icon = {
				image = "[?v.GetTokenKey]"  # If array contains scope references
				# Or
				# frame = v  # If array contains frame numbers
			}
		}

		dynamic_lists = {
			LIST_NAME_grid = {
				array = ROOT.ARRAY_NAME
				change_scope = no  # Set to yes if array contains countries/states
				entry_container = "LIST_NAME_entry"
			}
		}

		ai_enabled = {
			always = no
		}
	}
}
